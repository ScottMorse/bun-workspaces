# CLI Usage

## Global Options

<div style={{ marginBottom: "3rem" }}>
{getCliGlobalOptionsContent().map((option) => (

<div key={"cli-global-option-" + option.optionName}>
<h3 id={`global-option-${option.optionName}`}>{option.title}</h3>

      <p>
        <code>{option.mainOption}</code> |{" "}
        <code>{option.shortOption}</code>{" "}
      </p>
      {option.values?.length ? (
        <p>
          Values: {option.values.map((value) => <code style={{ marginRight: "0.25rem" }}>{value}</code>)}
        </p>
      ) : (
        ""
      )}
      {option.defaultValue ? (
        <p>
          Default: <code>{option.defaultValue}</code>
        </p>
      ) : (
        ""
      )}
      <p>{option.description}</p>
      <h5><em>Examples:</em></h5>
      <div className="code-block"  >
      <div>
      <code><span className="code-block-prefix">$ </span><span className="code-comment"># alias bw="bunx bun-workspaces"</span></code>
      </div>
      <code><span className="code-block-prefix">$ </span></code>
        {option.examples.map((example, i) => (
          <div key={'example-' + 'cli-global-option-' + option.optionName + '-' + i}>
           <code className={example.trim().startsWith("#") ? "code-comment" : ""} key={example}><span className="code-block-prefix">$ </span>{example}</code>
          </div>
        ))}
      </div>
    </div>

))}

</div>

## Commands

{getCliProjectCommandsContent().map((command) => (

<div key={"cli-command-" + command.title}>
<h3>{command.title}</h3>

      <p>
        <code>{command.command}</code>
      </p>
      {command.aliases?.length ? (
        <p>
          Aliases: {command.aliases.map((value) => <code style={{ marginRight: "0.25rem" }}>{value}</code>)}
        </p>
      ) : (
        ""
      )}
      <p style={{ marginBottom: "0" }}>{command.description}</p>
      {Object.values(command.options)?.length ? (
        <div style={{ marginTop: "1rem" }}>
        <h5><em>Options:</em></h5>
        <div style={{ marginTop: "0.25rem", marginBottom: "0.75rem" }}>
          {Object.values(command.options).map((option) => (
           <div key={"cli-command-option-" + command.command + "-" + option.flag}>
            <p style={{ margin: "0" }}><code>{option.flag}</code> {option.description}</p>
           </div>
          ))}
        </div>
        </div>
      ) : (
        ""
      )}
      <h5><em>Examples:</em></h5>
      <div className="code-block"  >
      <div>
      <code><span className="code-block-prefix">$ </span><span className="code-comment"># alias bw="bunx bun-workspaces"</span></code>
      </div>
      <code><span className="code-block-prefix">$ </span></code>
        {command.examples.map((example, i) => (
          <div key={'example-' + 'cli-global-option-' + command.optionName + '-' + i}>
           <code key={example}><span className="code-block-prefix">$ </span><span className={example.trim().startsWith("#") ? "code-comment" : ""}>{example}</span></code>
          </div>
        ))}
      </div>
    </div>

))}
